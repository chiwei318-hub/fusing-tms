import express from "express";
import dotenv from "dotenv";
import db from "./db.js";
import { issueInvoice } from "./invoiceProvider.js";

dotenv.config();
const app = express();
app.use(express.json());

function orderNo() {
  const d = new Date();
  return `FY${d.getFullYear()}${String(d.getMonth()+1).padStart(2,"0")}${String(d.getDate()).padStart(2,"0")}${String(Math.floor(Math.random()*10000)).padStart(4,"0")}`;
}

function getCustomer(id) {
  return db.prepare(`SELECT * FROM customers WHERE id = ?`).get(id);
}

function getOrder(id) {
  return db.prepare(`SELECT * FROM orders WHERE id = ?`).get(id);
}

app.get("/", (req, res) => {
  res.json({ ok: true, name: "Fuyong TMS eInvoice" });
});

app.post("/customers", (req, res) => {
  const { name, tax_id = "", invoice_title = "", email = "", billing_type = "cash" } = req.body;

  const r = db.prepare(`
    INSERT INTO customers (name, tax_id, invoice_title, email, billing_type)
    VALUES (?, ?, ?, ?, ?)
  `).run(name, tax_id, invoice_title, email, billing_type);

  res.json({ ok: true, customer_id: r.lastInsertRowid });
});

app.post("/orders", (req, res) => {
  const { customer_id, cargo_name = "運費", amount } = req.body;
  const customer = getCustomer(customer_id);
  if (!customer) return res.status(404).json({ ok: false, message: "找不到客戶" });

  const no = orderNo();
  const r = db.prepare(`
    INSERT INTO orders (order_no, customer_id, cargo_name, amount, status)
    VALUES (?, ?, ?, ?, 'created')
  `).run(no, customer_id, cargo_name, amount);

  res.json({ ok: true, order: getOrder(r.lastInsertRowid) });
});

app.post("/orders/:id/dispatch", (req, res) => {
  const order = getOrder(req.params.id);
  if (!order) return res.status(404).json({ ok: false, message: "找不到訂單" });

  db.prepare(`UPDATE orders SET status = 'dispatched' WHERE id = ?`).run(order.id);

  db.prepare(`
    INSERT INTO ar_ledger (customer_id, order_id, entry_type, amount, note)
    VALUES (?, ?, 'dispatch_cost', ?, ?)
  `).run(order.customer_id, order.id, Math.round(order.amount * 0.7), `派車成本 ${order.order_no}`);

  res.json({ ok: true, message: "已派車並記帳" });
});

app.post("/orders/:id/complete", async (req, res) => {
  const order = getOrder(req.params.id);
  if (!order) return res.status(404).json({ ok: false, message: "找不到訂單" });

  const customer = getCustomer(order.customer_id);
  if (!customer) return res.status(404).json({ ok: false, message: "找不到客戶" });

  if (customer.billing_type === "monthly") {
    db.prepare(`UPDATE orders SET status = 'monthly_pending' WHERE id = ?`).run(order.id);

    db.prepare(`
      INSERT INTO ar_ledger (customer_id, order_id, entry_type, amount, note)
      VALUES (?, ?, 'receivable', ?, ?)
    `).run(customer.id, order.id, order.amount, `月結應收 ${order.order_no}`);

    return res.json({ ok: true, message: "月結客戶，已掛應收，待月底開票" });
  }

  const invoice = await issueInvoice({ customer, order });

  const inv = db.prepare(`
    INSERT INTO invoices (order_id, customer_id, provider, invoice_no, random_no, amount, status, raw_response)
    VALUES (?, ?, ?, ?, ?, ?, 'issued', ?)
  `).run(
    order.id,
    customer.id,
    invoice.provider,
    invoice.invoiceNo,
    invoice.randomNo,
    order.amount,
    JSON.stringify(invoice.raw)
  );

  db.prepare(`
    UPDATE orders SET status = 'invoiced', invoice_id = ?
    WHERE id = ?
  `).run(inv.lastInsertRowid, order.id);

  db.prepare(`
    INSERT INTO ar_ledger (customer_id, order_id, entry_type, amount, note)
    VALUES (?, ?, 'receivable', ?, ?)
  `).run(customer.id, order.id, order.amount, `現結應收 ${order.order_no}`);

  res.json({
    ok: true,
    message: "訂單完成，自動開立電子發票",
    invoice_no: invoice.invoiceNo,
    random_no: invoice.randomNo
  });
});

app.post("/monthly/checkout", async (req, res) => {
  const { customer_id } = req.body;
  const customer = getCustomer(customer_id);
  if (!customer) return res.status(404).json({ ok: false, message: "找不到客戶" });

  const orders = db.prepare(`
    SELECT * FROM orders
    WHERE customer_id = ? AND status = 'monthly_pending'
  `).all(customer_id);

  if (!orders.length) return res.json({ ok: true, message: "沒有待開票月結單" });

  const total = orders.reduce((sum, x) => sum + x.amount, 0);

  const fakeOrder = {
    order_no: `MONTHLY-${customer_id}-${Date.now()}`,
    amount: total
  };

  const invoice = await issueInvoice({ customer, order: fakeOrder });

  db.prepare(`
    INSERT INTO invoices (order_id, customer_id, provider, invoice_no, random_no, amount, status, raw_response)
    VALUES (NULL, ?, ?, ?, ?, ?, 'issued', ?)
  `).run(
    customer.id,
    invoice.provider,
    invoice.invoiceNo,
    invoice.randomNo,
    total,
    JSON.stringify(invoice.raw)
  );

  db.prepare(`
    UPDATE orders SET status = 'invoiced'
    WHERE customer_id = ? AND status = 'monthly_pending'
  `).run(customer.id);

  res.json({
    ok: true,
    message: "月結批次開票完成",
    total,
    invoice_no: invoice.invoiceNo
  });
});

app.post("/payments", (req, res) => {
  const { customer_id, amount } = req.body;

  db.prepare(`
    INSERT INTO ar_ledger (customer_id, order_id, entry_type, amount, note)
    VALUES (?, NULL, 'payment', ?, ?)
  `).run(customer_id, -Math.abs(amount), `收款 ${amount}`);

  const row = db.prepare(`
    SELECT COALESCE(SUM(amount),0) AS balance
    FROM ar_ledger
    WHERE customer_id = ?
  `).get(customer_id);

  res.json({ ok: true, balance: row.balance });
});

app.listen(process.env.PORT || 3000, () => {
  console.log(`running on ${process.env.PORT || 3000}`);
});
